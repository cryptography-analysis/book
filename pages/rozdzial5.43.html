def find(qGram, qGrams):
    i = 0
    for j in range(30):
        if qGram == qGrams[j]:

def findT(triGram, triGrams):
    i = 0

        if triGram == triGrams[i]:
            i = j
            return i
        else:
            i = i+1
    return -1

def findB(biGram, biGrams):

    sum = 0;
    y = 0;
    qGram = ""
    triGram = ""
    biGram = ""

        qGram = ""

        y = find(qGram, qGrams)
        if y > 0:
            sum = sum + y
    for i in range(tL(text)-4):
        triGram = ""

        y = findT(triGram, triGrams)

            sum = sum + y

        y = findB(biGram, biGrams)
        if y > 0:
            sum = sum + y
    return sum
